      // Función para reproducir el video al hacer clic en la miniatura
      function playVideo() {
          var iframe = document.getElementById("videoIframe");
          var videoContainer = document.querySelector(".yt-mix-bg");
          
          videoContainer.style.display = "none"; // Ocultar la miniatura
          iframe.style.display = "block";  // Mostrar el iframe del video
          
          // Verifica que el video ya tiene el autoplay
          var iframeSrc = iframe.src;
          if (!iframeSrc.includes("autoplay=1")) {
              iframe.src = iframeSrc + "&autoplay=1";  // Iniciar la reproducción del video
          }
      }
    </script>
